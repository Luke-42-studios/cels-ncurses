```c
/* ----------------------------------------------------------------------------
 * Entity Component Set/Get (outside composition context)
 * ---------------------------------------------------------------------------- */

void cels_entity_set_component(cels_entity_t entity, cels_entity_t component_id,
                                const void* data, size_t size) {
    if (cels_context == NULL) return;
    ecs_world_t* world = cels_world_manager_world(cels_context->world);
    ecs_set_id(world, (ecs_entity_t)entity, (ecs_entity_t)component_id, size, data);
}

const void* cels_entity_get_component(cels_entity_t entity, cels_entity_t component_id) {
    if (cels_context == NULL) return NULL;
    ecs_world_t* world = cels_world_manager_world(cels_context->world);
    return ecs_get_id(world, (ecs_entity_t)entity, (ecs_entity_t)component_id);
}
```

Note: `cels_api.c` already includes `<flecs.h>` (it uses `ecs_set_id` in the REST config and `ecs_get_id` in `cels_watch_component`). The `cels_world_manager_world()` pattern is used throughout this file -- follow it exactly. The `cels_context` null-guard pattern is used by every function in this file.
