**Rewrite examples/minimal.c** as the Phase 1 validation example, following 01-RESEARCH.md Pattern 5 (Consumer Application Usage).

The CELS v0.5 consumer API uses these macros (verified from RESEARCH.md sources section referencing cels.h lines 606, 727, 782, 897):
- `cels_main()` -- entry point macro (cels.h line 727)
- `cels_scope { ... }` -- world configuration block
- `cels_session(Name) { ... }` -- session with composition (cels.h line 782)
- `cels_running()` -- main loop condition
- `cels_step(0)` -- advance one frame
- `CEL_Composition(Name) { ... }` -- entity composition declaration
- `cel_entity(.name = "X") { ... }` -- create named entity
- `cel_has(Type, ...)` -- attach component with designated initializers (cels.h line 897)

Write the following (this is the exact code, not conditional):

```c
// Phase 1 validation: NCurses module loads, window entity opens terminal
// Demonstrates: MOD-01, MOD-02, MOD-03, WIN-01, WIN-02, WIN-03
#include <cels/cels.h>
#include <cels-ncurses/ncurses.h>
#include <stdio.h>

// Root composition: create window entity with config
CEL_Composition(World) {
    (void)props;

    cel_entity(.name = "Window") {
        cel_has(NCurses_WindowConfig,
            .title = "Phase 1 Test",
            .fps = 30,
            .color_mode = 0  // auto
        );
    }
}

cels_main() {
    cels_scope {
        cels_world->title = "Phase 1 Module Test";
    }

    // Load NCurses module (registers components, systems, observers)
    NCurses_init();

    cels_register(World);

    cels_session(World) {
        // WIN-03: Query NCurses_WindowState for dimensions and running status
        // After first cels_step, the window observer has fired and attached state
        cels_step(0);

        // Find the window entity and read its state
        ecs_world_t* world = cels_get_world(cels_get_context());
        ecs_entity_t window = ecs_lookup(world, "Window");
        if (window) {
            const NCurses_WindowState* state = ecs_get_id(
                world, window, NCurses_WindowState_id);
            if (state) {
                // Print to stderr (ncurses owns stdout)
                fprintf(stderr, "[minimal] Window: %dx%d, running=%d\n",
                        state->width, state->height, state->running);
            }
        }

        while (cels_running()) {
            cels_step(0);
        }
    }
}
```
